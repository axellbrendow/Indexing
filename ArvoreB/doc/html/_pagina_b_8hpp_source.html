347  Tripla promoverElementoPara(Pagina* pagina, int indice, Pagina* paginaCheia,
348  Pagina* novaPagina, bool promoverElementoDoInicio)
349  {
350  Tripla tripla(

352  pagina->dados[indice], // Salva o dado no índice informado
353  pagina->ponteiros[indice + 1] // Salva o ponteiro posterior ao índice informado

358  transferirElementoPara(pagina, indice, indiceDeRemocao);
359 
360  numeroDeElementos--;
361 
362  // Quando um elemento é promovido, o ponteiro da esquerda dele deve apontar
363  // para a página que estava cheia (a que provocou a duplicação).
364  // Já o ponteiro da direita deve apontar para a nova página (gerada pela
365  // duplicação).
366  pagina->ponteiros[indice] = paginaCheia->endereco;
367  pagina->ponteiros[indice + 1] = novaPagina->endereco;
