350  void promoverElementoPara(
351  Pagina *paginaDestino, int indice, Pagina *exPaginaCheia,
352  Pagina *novaPagina, bool promoverElementoDoFim, fstream &arquivo)
353  {

360  }
361 
362  else // A página nova será afetada na promoção

365 
366  transferirElementoPara(paginaDestino, indice, 0);
367  }
368 
369  numeroDeElementos--;
370 
371  // atualiza as páginas no arquivo
372  exPaginaCheia->colocarNoArquivo(arquivo);
373  novaPagina->colocarNoArquivo(arquivo);
374 
375  // Quando um elemento é promovido, o ponteiro da esquerda dele deve apontar
376  // para a página que estava cheia (a que provocou a divisão).
377  // Já o ponteiro da direita deve apontar para a nova página (gerada pela
378  // divisão).

380  paginaDestino->ponteiros[indice + 1] = novaPagina->endereco;
381  paginaDestino->colocarNoArquivo(arquivo);
