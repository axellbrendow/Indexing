320  pair<Pagina *, bool> inserir(
321  TIPO_DAS_CHAVES chave, TIPO_DOS_DADOS dado,
322  int indiceDePromocao, file_pointer_type enderecoPaginaFilha,
323  file_pointer_type enderecoPaginaPai)
324  {
325  pair<Pagina *, bool> infoPai(nullptr, false);
326 
327  // Checa se a página foi carregada
328  if (carregar(paginaFilha, enderecoPaginaFilha))
329  {
330  // Checa se não existe um endereço para a página pai. Quando ele
331  // não existir é porque a recursividade voltou à raiz.
332  bool voltouParaARaiz =
333  enderecoPaginaPai == constantes::ptrNuloPagina;
334 
335  int indiceDeInsercao = paginaFilha->obterIndiceDeDescida(chave);
336 
337  // Checa se não há ponteiro de descida. Caso não, a página é uma folha.
338  if (paginaFilha->ponteiros[indiceDeInsercao] == constantes::ptrNuloPagina)
339  {
340  // Checa se a inserção teve sucesso
341  if (paginaFilha->inserir(chave, dado, indiceDeInsercao))
342  {
343  paginaFilha->colocarNoArquivo(arquivo);
344  }
345 
346  else // Inserção na página falhou, acontece quando ela está cheia.
347  {
348  tratarPaginaCheia(chave, dado, indiceDePromocao, infoPai);
349 
350  if (voltouParaARaiz) trocarRaizPor(paginaPai);
351  }
352  }
353 
354  else // A página não é uma folha, continuar descendo
355  {
356  paginaPai = paginaFilha;
357 
358  // O inserir retorna informações sobre a promoção de elementos para
359  // esta página
360  infoPai = inserir(
361  chave, dado, indiceDeInsercao,
362  paginaFilha->ponteiros[indiceDeInsercao], enderecoPaginaFilha);
363 
364  // Checa se esta página precisa promover algum elemento que tenha
365  // sido adicionado nela pelas suas filhas
366  if (infoPai.first != nullptr)
367  {
368  Pagina *paginaDeInsercao = infoPai.first;
369  bool inseriuNaPaginaFilha = infoPai.second;
370 
371  // As páginas que eram pais, na volta da recursividade, são filhas
372  paginaFilha = paginaPai;
373  paginaIrma = paginaIrmaPai;
374 
375  if (voltouParaARaiz) paginaPai->limpar();
376  else carregar(paginaPai, enderecoPaginaPai);
377 
378  promoverOParQueEstiverSobrando(
379  indiceDePromocao, paginaDeInsercao,
380  inseriuNaPaginaFilha, infoPai);
381 
382  if (voltouParaARaiz) trocarRaizPor(paginaPai);
383  }
384  }
385  }
386 
387  return infoPai;
388  }
