417  void inserir(TIPO_DAS_CHAVES &chave, TIPO_DOS_DADOS &dado,
418  list<file_ptr_type>& pilhaDeEnderecos,
419  list<int>& pilhaDeIndices)
420  {
421  // Após a função obterCaminhoDeDescida(), paginaFilha aponta para a última
422  // página carregada na descida da árvore (alguma folha).
423  int indiceDeInsercao = paginaFilha->obterIndiceDeDescida(chave);
424 
425  // Checa se a inserção teve sucesso
426  if (paginaFilha->inserir(chave, dado, indiceDeInsercao))
427  paginaFilha->colocarNoArquivo(arquivo);
428 
429  else // Inserção na página falhou, acontece quando ela está cheia.
430  {
431  pilhaDeEnderecos.pop_back(); // O último endereço pode ser descartado
432  bool voltouParaARaiz = pilhaDeEnderecos.empty();
433  int indiceDePromocao = pilhaDeIndices.back();
434 
435  pair<Pagina *, bool> infoPai(nullptr, false);
436  tratarPaginaCheia(chave, dado, indiceDePromocao, infoPai);
437 
438  if (voltouParaARaiz) trocarRaizPor(paginaPai);
439 
440  else // Inicia o processo de subida na árvore
441  {
442  // Quando uma página que é filha da atual promove um elemento para
443  // a atual ou para a irmã da atual, a variável paginaDeInsercao vai
444  // ter um ponteiro para a página que receber o elemento promovido.
445  Pagina *paginaDeInsercao = infoPai.first;
446  bool inseriuNaPaginaAtual;
447  file_ptr_type enderecoPaginaPai;
448 
449  // Checa se houve alguma promoção para a página atual
450  while (paginaDeInsercao != nullptr)
451  {
452  pilhaDeEnderecos.pop_back();
453  voltouParaARaiz = pilhaDeEnderecos.empty();
454  enderecoPaginaPai = voltouParaARaiz ?
455  constantes::ptrNuloPagina : pilhaDeEnderecos.back();
456 
457  pilhaDeIndices.pop_back();
458  indiceDePromocao = pilhaDeIndices.back();
459  inseriuNaPaginaAtual = infoPai.second;
460 
461  // As páginas que eram pais, na subida da árvore, são filhas
462  swap(paginaPai, paginaFilha);
463  swap(paginaIrmaPai, paginaIrma);
464 
465  if (voltouParaARaiz)
466  {
467  paginaPai->limpar();
468  paginaPai->ponteiros.push_back(constantes::ptrNuloPagina);
469  }
470 
471  else carregar(paginaPai, enderecoPaginaPai);
472 
473  promoverOParQueEstiverSobrando(
474  indiceDePromocao, paginaDeInsercao,
475  inseriuNaPaginaAtual, infoPai);
476 
477  if (voltouParaARaiz) trocarRaizPor(paginaPai);
478 
479  paginaDeInsercao = infoPai.first;
480  }
481  }
482  }
